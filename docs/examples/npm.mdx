### Check that a package has specific scripts

```json .commonality/config.json
import * as recommended from 'commonality-checks-recommended';

export default defineConfig({
  checks: {
    buildable: [
      recommended.hasJsonFile('package.json', {
        scripts: {
          build: 'tsc --build',
          dev: 'tsc --watch',
          clean: 'rimraf dist',
        },
      }),
    ],
  },
});
```

### Check that a package has an `.npmignore`

```json .commonality/config.json
import * as recommended from 'commonality-checks-recommended';

export default defineConfig({
  checks: {
    '*': [
      recommended.hasTextFile('.npmignore', [
        'node_modules',
        'dist',
        'coverage',
      ]),
    ],
  },
});
```
