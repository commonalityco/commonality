import { hasTextFile } from 'commonality-checks-recommended';

  'export default defineConfig({',
  '  server: {',
  '    port: 3000,',
  '  },',
  '  build: {',
  "    outDir: 'dist',",
  '  },',
  '  resolve: {',
  '    alias: {',
  "      '@': '/src',",
  '    },',
  '  },',
  '})',
]);
